113 void turnAngle(double angle) {
114  const double rawTarget = degreesToEdges(angle);
115 
116  const double lStart = lEnc.read();
117  const double rStart = rEnc.read();
118 
119  const double lTarget = -rawTarget + lEnc.read();
120  const double rTarget = rawTarget + rEnc.read();
121 
122  // Using signs like this allows for using the same condition regardless of turning CW or CCW--It works, I promise
123  while (signum(lEnc.read() - lTarget) * signum(rawTarget) >= 0 && signum(rEnc.read() - rTarget) * signum(rawTarget) <= 0) {
124  const double ramp = bothRamp(lStart, lTarget, lEnc.read());
125 
126  lSetpoint = (0.07 + ramp * 0.3) * ((angle < 0.0) ? 1 : -1);
127  rSetpoint = -lSetpoint;
128 
129  updatePID();
130

132 
133  delay(5);

142 void updatePID() {
143  double t = millis();
144 
145  if (t - lastPIDUpdate < 5) {
146  return;
147  }
148 
149  long a = lEnc.read();
150  long b = rEnc.read();
151 
152  // da and db are the velocity error values in edges per millisecond
153  double da = (a - lLastPos) / (t - lastPIDUpdate) - lSetpoint * 7.0;
154  double db = (b - rLastPos) / (t - lastPIDUpdate) - rSetpoint * 7.0;
155 
156  lLastPos = a;
157  rLastPos = b;
158 
159  // Feedforward and proportional gains for each wheel
160  const double lF = 7.0 / 8.0, lP = 0.4;
161  const double rF = 7.0 / 7.6, rP = 0.4;
162 
163  lOutput = lSetpoint * lF - da * lP;
164  rOutput = rSetpoint * rF - db * rP;
165 
166  lastPIDUpdate = millis();
167 }
