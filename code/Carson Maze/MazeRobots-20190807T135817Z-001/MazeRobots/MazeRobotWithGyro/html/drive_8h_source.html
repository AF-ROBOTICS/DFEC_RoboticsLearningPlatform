237 void turnAngle(double angle) {
238  const double rawTarget = degreesToEdges(angle);
239 
240  const double lStart = lEnc.read();
241  const double rStart = rEnc.read();
242 
243  const double lTarget = -rawTarget + lEnc.read();
244  const double rTarget = rawTarget + rEnc.read();
245 
246  // Using signs like this allows for using the same condition regardless of turning CW or CCW--It works, I promise
247  while (signum(lEnc.read() - lTarget) * signum(rawTarget) >= 0 && signum(rEnc.read() - rTarget) * signum(rawTarget) <= 0) {
248  const double ramp = bothRamp(lStart, lTarget, lEnc.read());
249 
250  lSetpoint = (0.07 + ramp * 0.3) * ((angle < 0.0) ? 1 : -1);
251  rSetpoint = -lSetpoint;
252 
253  updatePID();
254 
255  drive(255 * lOutput, 255 * rOutput);
256 
257  delay(5);
258  }
259 
260  drive(0, 0);
261 }

266 void updatePID() {
267  double t = millis();
268 
269  if (t - lastPIDUpdate < 5) {
270  return;
271  }
272 
273  long a = lEnc.read();
274  long b = rEnc.read();
275 
276  // da and db are the velocity error values in edges per millisecond
277  double da = (a - lLastPos) / (t - lastPIDUpdate) - lSetpoint * 7.0;
278  double db = (b - rLastPos) / (t - lastPIDUpdate) - rSetpoint * 7.0;
279 
280  lLastPos = a;
281  rLastPos = b;
282 
283  // Feedforward and proportional gains for each wheel
284  const double lF = 7.0 / 8.0, lP = 0.4;
285  const double rF = 7.0 / 7.6, rP = 0.4;
286 
287  lOutput = lSetpoint * lF - da * lP;
288  rOutput = rSetpoint * rF - db * rP;
289 
290  lastPIDUpdate = millis();
291 }
